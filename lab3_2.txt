#include <iostream>
#include <fstream>
#include <algorithm>
#include <string>
int main()
{
	setlocale(LC_ALL, "Russian");
	std::string path;
	char text[256];
	path = "D:\\Study\\TextFile1.txt"; //”казываем путь
	std::ifstream file(path);
	file.getline(text, 256);
	std::cout << text<<std::endl;
	int Start = 0;
	int Finish = 0;
	int C_Buff = 0;
	int C = 0;
	int Buff_Start = 0;
	int Buff_Finish = 0;
	for (int i = 0; text[i] != '\0'; i++)
	{
		C_Buff = 0;
		if (text[i] != ' ')
		{
			Buff_Start = i;
			while (text[i] != ' ')
			{																																																								
				i++;
				C_Buff++;
				if (text[i] =='?')
					break;
			}
			Buff_Finish = i;
			if (C < C_Buff)
			{
				C = C_Buff;
				Start = Buff_Start;
				Finish = Buff_Finish;
			}
		}
	}
	for (Start; Start < Finish; Start++)
	{
		std::cout << text[Start];
	}
}
