#include <iostream>
#include <fstream>
#include <string>
int main()
{
	setlocale(LC_ALL, "Russian");
	int* mass;
	int C = 0, Buff, K;
	std::string S, O;
	std::cout << "Введите путь к файлу";
	std::cin >> S;
	std::string path = S;
	std::ifstream fin;
	fin.open(path);
	if (!fin.is_open())
	{
		std::cout << "Ошибка открытия файла!\n";
		return 0;
	}
	while (!fin.eof())
	{
		O = " ";//подсчет чисел, на место вставляет данные
		fin >> O;
		C++;
	}
	fin.close();
	fin.open(path);
	mass = new int[C];
	for (int i = 0; i < C; i++)
	{
		fin >> K;
		mass[i] = K;
	}
	fin.close();
	for (int i = 0; i < C; i++)
	{
		for (int P = C - 1; P > i; P--)
			if (mass[P] > mass[P - 1])
			{
				Buff = mass[P];
				mass[P] = mass[P - 1];
				mass[P - 1] = Buff;
			}
	}
	for (int i = 0; i < C; i++)
	{
		if (mass[i] % 2 == 0)
			std::cout << mass[i] << "\n";
	}
}
