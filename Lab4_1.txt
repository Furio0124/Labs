#include <iostream>
float Rekur(float A[], int n)
{
	if (n == 1) return A[0];
	else return Rekur(A, n - 1) * (n - 1) / n + A[n-1] / n;
}

int main()
{
	setlocale(LC_ALL, "Russian");
	int N,K=0;
	int Sum = 0;
	int Z=0;
	cout << "Введите N";
	cin >> N;
	float* A = new float[N];
	cout << "Введите значение переменных\n";
	for (int i = 0; i < N; i++)
	{
		cout << "A[" << i << "]";
		cin >> K;
		A[i] = K;
	}
	cout << Rekur(A,N);
}
