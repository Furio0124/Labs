#include <iostream>
using namespace std;
int slojenie(int X,int Y, int C)
{
	if ((X >= 0) && (Y > 0))
	{
		Y--;
		X++;
		return slojenie(X, Y, C);
	}
	else if ((X >= 0) && (Y < 0))
	{
		Y++;
		X--;
		return slojenie(X, Y, C);
	}
	else if ((X <= 0) && (Y > 0))
	{
		Y--;
		X++;
		return slojenie(X, Y, C);
	}
	else if ((X <= 0) && (Y < 0))
	{
		Y++;
		X--;
		return slojenie(X, Y, C);
	}
	else if (Y == 0)
	{
		return X;
	}
}
int main()
{
	setlocale(LC_ALL, "Russian");
	int X = 0, Y = 0,C=0;
	cout << "Введите число X\n";
	cin >> X;
	cout << "Введите число Y\n";
	cin >> Y;
	cout << X << "+" << Y << "=" << slojenie(X, Y,C);
}
